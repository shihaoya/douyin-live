// src/websocket/douyin.js
const WebSocket = require('ws');
const EventEmitter = require('events');
const logger = require('../config/logger');

class DouyinWebSocket extends EventEmitter {
  constructor(roomId, options = {}) {
    super();
    this.roomId = roomId;
    this.ws = null;
    this.isConnected = false;
    this.reconnectAttempts = 0;
    this.maxReconnectAttempts = options.maxReconnectAttempts || 10;
    this.reconnectInterval = options.reconnectInterval || 5000;
  }

  async connect() {
    try {
      // 1. 获取wss地址
      const wssUrl = await this.getWssUrl();
      
      // 2. 建立WebSocket连接
      this.ws = new WebSocket(wssUrl);
      
      // 3. 绑定事件
      this.ws.on('open', () => this.handleOpen());
      this.ws.on('message', (data) => this.handleMessage(data));
      this.ws.on('close', () => this handleClose());
      this.ws.on('error', (error) => this.handleError(error));
      
    } catch (error) {
      logger.error(`连接失败: ${error.message}`);
      this.reconnect();
    }
  }

  handleOpen() {
    logger.info(`直播间 ${this.roomId} 连接成功`);
    this.isConnected = true;
    this.reconnectAttempts = 0;
    this.emit('connected', this.roomId);
    
    // 启动心跳
    this.startHeartbeat();
  }

  handleMessage(data) {
    try {
      // 解析Protobuf消息
      const message = this.parseProtobuf(data);
      
      // 发送解析后的消息
      this.emit('message', {
        roomId: this.roomId,
        ...message
      });
      
    } catch (error) {
      logger.error(`消息解析失败: ${error.message}`);
    }
  }

  handleClose() {
    logger.warn(`直播间 ${this.roomId} 连接关闭`);
    this.isConnected = false;
    this.stopHeartbeat();
    this.emit('disconnected', this.roomId);
    
    // 尝试重连
    this.reconnect();
  }

  reconnect() {
    if (this.reconnectAttempts < this.maxReconnectAttempts) {
      this.reconnectAttempts++;
      logger.info(`${this.reconnectAttempts}秒后尝试重连...`);
      
      setTimeout(() => {
        this.connect();
      }, this.reconnectInterval);
    } else {
      logger.error('达到最大重连次数，停止重连');
      this.emit('reconnect_failed', this.roomId);
    }
  }

  disconnect() {
    if (this.ws) {
      this.ws.close();
    }
  }

  // 私有方法
  async getWssUrl() {

  }

  parseProtobuf(buffer) {

  }

  startHeartbeat() {
    // 实现心跳逻辑
  }

  stopHeartbeat() {

  }
}

module.exports = DouyinWebSocket;

```javascript
// src/services/templateEngine.js

/**
 * 安全的对象属性访问
 */
function safeGet(obj, path) {
  if (!obj || !path) return undefined;
  
  return path.split('.').reduce((acc, part) => {
    return acc && acc[part] !== undefined ? acc[part] : undefined;
  }, obj);
}

/**
 * 解析并评估模板表达式
 */
function evaluateTemplate(template, data) {
  try {
    // 简化版本：支持 || 回退和 + 拼接
    
    // 1. 处理括号内的表达式
    template = template.replace(/\(([^)]+)\)/g, (match, expr) => {
      return evaluateExpression(expr.trim(), data);
    });
    
    // 2. 处理顶层表达式
    return evaluateExpression(template, data);
    
  } catch (error) {
    console.error('模板解析错误:', error);
    return '异常';
  }
}

/**
 * 评估单个表达式
 */
function evaluateExpression(expr, data) {
  // 检查是否包含 || （回退逻辑）
  if (expr.includes('||')) {
    const parts = expr.split('||').map(p => p.trim());
    
    for (const part of parts) {
      const result = evaluateSingle(part, data);
      if (result !== undefined && result !== null && result !== '') {
        return result;
      }
    }
    
    return ''; // 所有回退都失败
  }
  
  // 没有回退，直接评估
  return evaluateSingle(expr, data);
}

/**
 * 评估简单表达式（路径访问或字符串）
 */
function evaluateSingle(expr, data) {
  // 如果是字符串字面量
  if ((expr.startsWith('"') && expr.endsWith('"')) ||
      (expr.startsWith("'") && expr.endsWith("'"))) {
    return expr.slice(1, -1);
  }
  
  // 如果是数字
  if (/^\d+$/.test(expr)) {
    return parseInt(expr);
  }
  
  // 否则作为路径访问
  return safeGet(data, expr);
}

/**
 * 生成描述（主函数）
 */
function generateDescription(template, rawData) {
  try {
    const result = evaluateTemplate(template, rawData);
    return result || '异常';
  } catch (error) {
    console.error('描述生成失败:', error);
    return '异常';
  }
}

module.exports = {
  generateDescription,
  evaluateTemplate
};
```
